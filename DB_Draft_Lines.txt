// Gang_Member Section 
TABLE Gang_Member {
  ID_GM INT pk
  GM_name VARCHAR(40) 
  Job VARCHAR(40)
  Leader INT 
}

// Sub_Group Section 
TABLE Sub_Group {
  ID_SG INT pk
  SG_Name VARCHAR(40)
}

// GS Section 
TABLE GS {
  ID_GS INT pk
  ID_GM INT pk
  ID_SG INT pk
} 

// GS Section 
TABLE V_GS {
  ID_VGS INT pk
  ID_GS INT pk
  ID_Victim INT pk
} 

// Territory Section 
TABLE Territory {
  ID_Territory INT pk
  Territory_Name VARCHAR(40)
}

// Victim Section 
TABLE Victim {
  ID_Victim INT pk
  ID_GS INT
  Age INT 
  Blood_Type VARCHAR(3)
  ID_Territory INT
  Date_Of_Murder TIMESTAMP
}

// Parts_Name Section 
TABLE P_V {
  ID_PV INT pk
  ID_Part INT 
  ID_Victim INT
  ID_Sales VARCHAR(40)
} 

// Parts_Name Section 
TABLE Parts_Name {
  ID_Part INT pk
  Part_Name VARCHAR(40) 
  Available VARCHAR(1)
} 

// Dealer Section 
TABLE Dealer {
  ID_Dealer INT pk
  Dealer_Name VARCHAR(40)
}

// Customers Section 
TABLE Customers {
  ID_Customer INT pk
  Customer_Name VARCHAR(40)
  Age INT 
  Blood_Type VARCHAR(3)
  Wanted_Part VARCHAR(40)
}

// Sales Section 
TABLE Sales {
  ID_Sales INT pk
  ID_Customer INT
  ID_Dealer INT
  Price INT 
}
